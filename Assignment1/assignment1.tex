\section*{Question 7.2 }
Use a graph paper for this question (Take $2 cms = 1 unit$ on both x and y axis)
\begin{enumerate}[ label=(\arabic*)]
	\item Plot the following points: A(0,4), B(2,3), C(1,1) and D(2,0).
	\item Reflect points B, C, D on the y-axis and write down their coordinates. Name
	the images as B', C', D' respectively.
	\item Join the points A, B, C, D, D', C', B' and A in order, so as to form a closed
	figure. Write down the equation of the line of symmetry of the figure formed.
\end{enumerate}      

\section*{Solution:}
\begin{enumerate}[label=(\arabic*)]
	\item the plot of all points in last plot section labebled with  A,B,C,D\\
	\item This is general formula for image of point P in line with equation:
	\begin{align*}
	 \vec{n}^{\top}\vec{x} = c
	\end{align*}
	\begin{align*}
		\vec{R} &=
		\vec{P} + 2\frac{c - \vec{n}^{\top}\vec{P}}{\norm{\vec{n}}^2}\vec{n}
	\end{align*}
 Here in our problem
 \begin{align*}
 	\vec{B}&=\myvec{2\\3} ,
 	\vec{C}=\myvec{1\\1} ,
 	\vec{D}=\myvec{2\\0}\\
 	\vec{n}&=\myvec{1\\0}, c=0
	\end{align*}
   Substitute B,C,D in the formula\\
   \begin{align*}
   		\vec{B'} &=
   	\vec{B} + 2\frac{c - \vec{n}^{\top}\vec{B}}{\norm{\vec{n}}^2}\vec{n}
   	\\
   	    &=
    \myvec{2\\3} + 2\frac{0 - \myvec{1&0}\myvec{2\\3}}{1^2}\myvec{1\\0}
    \\
    	&=\myvec{2\\3}-\myvec{4\\0}
     \\
     	&=\myvec{-2\\3}
   \end{align*}
'similarly C' and D' then we get
\begin{align*}
		\vec{B'}=\myvec{-2\\3}, 
	\vec{C'}=\myvec{-1\\1}
	\vec{D'}=\myvec{-2\\0}
	\end{align*}
	\item joining the points in order  A, B, C, D, D', C', B' and A gives a polygon which is shown below.As u can see y axis cuts the figure equally so line of symmetry equation:
	\begin{align*}
		x=0.
	\end{align*}
	
\end{enumerate}
